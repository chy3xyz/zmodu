文件首行含 `//! ⛔ @generated by zmodu — DO NOT EDIT` → **跳过，禁止修改**。

```
禁止编辑: model.zig | persistence.zig | service.zig | api.zig | module.zig | main.zig | build.zig | build.zig.zon
```

需改结构时：改 schema.sql → `zmodu scaffold --force`。

## 阶段 3 — 逐项实现

### 3.1 业务规则 → ext/service.zig

```zig
// src/modules/users/ext/service.zig
const std = @import("std");
const zigmodu = @import("zigmodu");
const svc = @import("../service.zig");
const model = @import("../model.zig");

pub const UsersServiceExt = struct {
    svc: *svc.UsersService,
    backend: zigmodu.data.SqlxBackend,

    pub fn init(s: *svc.UsersService, b: zigmodu.data.SqlxBackend) @This() {
        return .{ .svc = s, .backend = b };
    }

    /// 注册 — 检查邮箱唯一性
    pub fn register(self: *@This(), entity: model.Users) !model.Users {
        // 检查邮箱是否已存在
        var repo = self.svc.persistence.usersRepo();
        const existing = try repo.findOne("email = ?", &.{.{ .string = entity.email }});
        if (existing != null) return error.EmailAlreadyExists;
        return self.svc.createUsers(entity);
    }

    /// 登录验证
    pub fn authenticate(self: *@This(), email: []const u8, password: []const u8) !model.Users {
        _ = password; // TODO: hash check
        var repo = self.svc.persistence.usersRepo();
        const user = try repo.findOne("email = ?", &.{.{ .string = email }});
        if (user == null) return error.InvalidCredentials;
        return user.?;
    }

    /// 管理员检查
    pub fn isAdmin(self: *@This(), user_id: i64) !bool {
        const user = try self.svc.getUsers(user_id);
        if (user == null) return false;
        return std.mem.eql(u8, user.?.role, "admin");
    }
};
```

### 3.2 自定义端点 → ext/api.zig

```zig
// src/modules/users/ext/api.zig
const std = @import("std");
const http = @import("zigmodu").http;
const ext = @import("service.zig");

pub const UsersApiExt = struct {
    ext: *ext.UsersServiceExt,
    pub fn init(e: *ext.UsersServiceExt) @This() { return .{ .ext = e }; }

    pub fn registerRoutes(self: *@This(), group: *http.RouteGroup) !void {
        try group.post("/auth/login", loginHandler, @ptrCast(@alignCast(self)));
        try group.get("/users/me", meHandler, @ptrCast(@alignCast(self)));
    }

    fn loginHandler(ctx: *http.Context) !void {
        const self: *@This() = @ptrCast(@alignCast(ctx.user_data orelse unreachable));
        const LoginReq = struct { email: []const u8, password: []const u8 };
        const req = ctx.bindJson(LoginReq) catch {
            try ctx.json(400, "{\"error\":\"invalid body\"}");
            return;
        };
        const user = self.ext.authenticate(req.email, req.password) catch {
            try ctx.json(401, "{\"error\":\"invalid credentials\"}");
            return;
        };
        try ctx.jsonStruct(200, user);
    }

    fn meHandler(ctx: *http.Context) !void {
        try ctx.json(200, "{\"user\":\"current\"}");
    }
};
```

### 3.3 状态机校验

```zig
// src/modules/orders/ext/service.zig
pub fn transitionStatus(self: *@This(), order_id: i64, new_status: []const u8) !void {
    const order = try self.svc.getOrders(order_id) orelse return error.NotFound;

    const valid = switch (order.status) {
        "pending" => std.mem.eql(u8, new_status, "confirmed") or
                     std.mem.eql(u8, new_status, "cancelled"),
        "confirmed" => std.mem.eql(u8, new_status, "shipped"),
        "shipped" => std.mem.eql(u8, new_status, "completed"),
        else => false,
    };
    if (!valid) return error.InvalidStatusTransition;

    try self.svc.updateOrders(.{ .id = order_id, .status = new_status });
}
```

## 快速参考

```
路由参数:  {id} 不是 :id
获取参数:  ctx.paramInt(T,"key")   ctx.queryInt(T,"key",default)
解析体:    ctx.bindJson(T) catch { ctx.json(400,...); return; }
响应:      ctx.jsonStruct(status, data)
成功:      http.RenderExt.success(ctx, "ok")
PK 字段:   id: ?i64 = null
可选字段:  email: ?[]const u8 = null
导入模块:  const x = @import("modules/x/module.zig")
子系统:    const shop_o = @import("modules/shop/order/module.zig")
错误:      return error.SomeBusinessError
事务:      repo.transact(R, struct { fn run(tx) !R { ... } }.run)
```
